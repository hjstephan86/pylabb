1""" 

2pylabb.core.state_space 

3====================== 

4Zustandsraumdarstellung ẋ = A·x + B·u, y = C·x + D·u. 

5 

6Unterstützt: 

7- Pole- / Eigenwert-Analyse 

8- Steuerbarkeit / Beobachtbarkeit (Grammsche Matrizen und Rang) 

9- Zeitantworten (Sprung, Impuls, beliebiger Eingang) 

10- Konvertierung in Übertragungsfunktion 

11- Diskretisierung per ZOH oder Euler 

12- Zustandsrückführung (Polvorgabe via scipy.signal.place_poles) 

13- LQR-Entwurf (scipy.linalg.solve_continuous_are) 

14""" 

15 

16from __future__ import annotations

18from typing import Optional, Sequence 

19import numpy as np 

20from numpy.typing import NDArray 

21import scipy.linalg as la 

22import scipy.signal as sig

25class StateSpace: 

26 """Lineare, zeitinvariante Zustandsraumdarstellung. 

27 

28 Parameters 

29 ---------- 

30 A : Systemmatrix (n×n). 

31 B : Eingangsmatrix (n×m). 

32 C : Ausgangsmatrix (p×n). 

33 D : Durchgangsmatrix (p×m). 

34 dt : Abtastzeit [s]; None → zeitkontinuierlich. 

35 name : Bezeichner. 

36 """ 

37 

38 def __init__( 

39 self, 

40 A: ArrayLike, 

41 B: ArrayLike, 

42 C: ArrayLike, 

43 D: ArrayLike, 

44 dt: Optional[float] = None, 

45 name: str = "SS", 

46 ) -> None: 

47 self.A = np.atleast_2d(np.asarray(A, dtype=float)) 

48 self.B = np.atleast_2d(np.asarray(B, dtype=float)) 

49 self.C = np.atleast_2d(np.asarray(C, dtype=float)) 

50 self.D = np.atleast_2d(np.asarray(D, dtype=float)) 

51 self.dt = dt 

52 self.name = name 

53 self._validate() 

54 

55 # ------------------------------------------------------------------ 

56 # Validierung 

57 # ------------------------------------------------------------------ 

58 

59 def _validate(self) -> None: 

60 n = self.A.shape[0] 

61 if self.A.shape != (n, n): 

62 raise ValueError(f"A muss quadratisch sein, hat Form {self.A.shape}.") 

63 if self.B.shape[0] != n: 

64 raise ValueError("B-Zeilen müssen A-Ordnung entsprechen.") 

65 if self.C.shape[1] != n: 

66 raise ValueError("C-Spalten müssen A-Ordnung entsprechen.") 

67 if self.D.shape != (self.C.shape[0], self.B.shape[1]): 

68 raise ValueError("D hat falsche Dimension.") 

69 

70 # ------------------------------------------------------------------ 

71 # Eigenschaften 

72 # ------------------------------------------------------------------ 

73 

74 @property 

75 def order(self) -> int: 

76 """Systemordnung n.""" 

77 return self.A.shape[0] 

78 

79 @property 

80 def n_inputs(self) -> int: 

81 """Anzahl Eingänge m.""" 

82 return self.B.shape[1] 

83 

84 @property 

85 def n_outputs(self) -> int: 

86 """Anzahl Ausgänge p.""" 

87 return self.C.shape[0] 

88 

89 # ------------------------------------------------------------------ 

90 # Stabilität / Analyse 

91 # ------------------------------------------------------------------ 

92 

93 def eigenvalues(self) -> NDArray: 

94 """Eigenwerte der Systemmatrix A.""" 

95 return np.linalg.eigvals(self.A) 

96 

97 def is_stable(self) -> bool: 

98 """Überprüft Stabilität anhand der Eigenwerte.""" 

99 ev = self.eigenvalues() 

100 if self.dt is None: 

101 return bool(np.all(np.real(ev) < 0)) 

102 return bool(np.all(np.abs(ev) < 1.0)) 

103 

104 def controllability_matrix(self) -> NDArray: 

105 """Steuerbarkeitsmatrix R = [B, AB, A²B, …, A^{n-1}B].""" 

106 n = self.order 

107 M = self.B.copy() 

108 Ak = np.eye(n) 

109 for _ in range(1, n): 

110 Ak = Ak @ self.A 

111 M = np.hstack([M, Ak @ self.B]) 

112 return M 

113 

114 def observability_matrix(self) -> NDArray: 

115 """Beobachtbarkeitsmatrix O = [C; CA; CA²; …; CA^{n-1}]ᵀ.""" 

116 n = self.order 

117 M = self.C.copy() 

118 Ak = np.eye(n) 

119 for _ in range(1, n): 

120 Ak = Ak @ self.A 

121 M = np.vstack([M, self.C @ Ak]) 

122 return M 

123 

124 def is_controllable(self) -> bool: 

125 """True, wenn der Rangtest der Steuerbarkeitsmatrix vollen Rang zeigt.""" 

126 Mc = self.controllability_matrix() 

127 return int(np.linalg.matrix_rank(Mc)) == self.order 

128 

129 def is_observable(self) -> bool: 

130 """True, wenn der Rangtest der Beobachtbarkeitsmatrix vollen Rang zeigt.""" 

131 Mo = self.observability_matrix() 

132 return int(np.linalg.matrix_rank(Mo)) == self.order 

133 

134 # ------------------------------------------------------------------ 

135 # Freqenzgang 

136 # ------------------------------------------------------------------ 

137 

138 def freqresp( 

139 self, omega: Optional[NDArray] = None, n_points: int = 500 

140 ) -> tuple[NDArray, NDArray]: 

141 """Frequenzgang H(jω) = C·(jωI−A)⁻¹·B + D.""" 

142 if omega is None: 

143 omega = np.logspace(-2, 2, n_points) 

144 n = self.order 

145 H_list = [] 

146 for w in omega: 

147 if self.dt is None: 

148 jw = 1j * w 

149 else: 

150 jw = np.exp(1j * w * self.dt) 

151 mat = jw * np.eye(n) - self.A 

152 H_list.append(self.C @ np.linalg.solve(mat, self.B) + self.D) 

153 H = np.array(H_list) # (n_omega, p, m) 

154 return omega, H

156 # ------------------------------------------------------------------ 

157 # Zeitantworten 

158 # ------------------------------------------------------------------ 

159 

160 def step_response( 

161 self, t_end: Optional[float] = None, dt: Optional[float] = None 

162 ) -> tuple[NDArray, NDArray]: 

163 """Sprungantwort. 

164 

165 Returns 

166 ------- 

167 t, y (y hat Form (n_t, p, m)) 

168 """ 

169 t_arr = self._sim_time(t_end, dt) 

170 if self.dt is None: 

171 t_out, y_out = sig.step( 

172 sig.lti(self.A, self.B, self.C, self.D), T=t_arr 

173 ) 

174 else: 

175 t_out, y_out = sig.dstep( 

176 sig.dlti(self.A, self.B, self.C, self.D, dt=self.dt), 

177 t=np.arange(int(t_arr[-1] / self.dt) + 1) * self.dt, 

178 ) 

179 y_out = np.squeeze(y_out) 

180 return t_out, y_out 

181 

182 def lsim(self, u: NDArray, t: NDArray) -> tuple[NDArray, NDArray]: 

183 """Lineare Simulation mit beliebigem Eingang. 

184 

185 Returns 

186 ------- 

187 t, y 

188 """ 

189 if self.dt is None: 

190 t_out, y_out, _ = sig.lsim( 

191 sig.lti(self.A, self.B, self.C, self.D), U=u, T=t 

192 ) 

193 else: 

194 t_out, y_out, _ = sig.dlsim( 

195 sig.dlti(self.A, self.B, self.C, self.D, dt=self.dt), 

196 u=u.reshape(-1, 1), t=t, 

197 ) 

198 y_out = y_out.flatten() 

199 return t_out, y_out

203 # ------------------------------------------------------------------ 

204 

205 def place_poles( 

206 self, desired_poles: Sequence[complex] 

207 ) -> NDArray: 

208 """Zustandsrückführung K durch Polvorgabe (nur SISO/MISO). 

209 

210 Returns 

211 ------- 

212 K : Rückführungsmatrix (m×n) so dass eig(A − B·K) ≈ desired_poles. 

213 """ 

214 if not self.is_controllable(): 

215 raise ValueError("System ist nicht vollständig steuerbar.") 

216 result = sig.place_poles(self.A, self.B, desired_poles) 

217 return result.gain_matrix 

218 

219 def lqr( 

220 self, 

221 Q: Optional[NDArray] = None, 

222 R: Optional[NDArray] = None, 

223 ) -> tuple[NDArray, NDArray]: 

224 """LQR-Reglerentwurf (Linear Quadratic Regulator). 

225 

226 Minimiert J = ∫(xᵀQx + uᵀRu) dt durch Lösen der 

227 algebraischen Riccati-Gleichung. 

228 

229 Parameters 

230 ---------- 

231 Q : Zustandswichtungsmatrix (n×n), Standard: Einheitsmatrix. 

232 R : Eingangswichtungsmatrix (m×m), Standard: Einheitsmatrix. 

233 

234 Returns 

235 ------- 

236 K : Optimale Verstärkungsmatrix (m×n). 

237 P : Lösung der Riccati-Gleichung. 

238 """ 

239 n = self.order 

240 m = self.n_inputs 

241 Q = Q if Q is not None else np.eye(n) 

242 R = R if R is not None else np.eye(m) 

243 P = la.solve_continuous_are(self.A, self.B, Q, R) 

244 K = np.linalg.solve(R, self.B.T @ P) 

245 return K, P 

246 

247 def observer_gain( 

248 self, desired_poles: Sequence[complex] 

249 ) -> NDArray: 

250 """Beobachter-Rückführung L per Polvorgabe (Luenberger-Beobachter). 

251 

252 Returns 

253 ------- 

254 L : Beobachtermatrix (n×p). 

255 """ 

256 if not self.is_observable(): 

257 raise ValueError("System ist nicht vollständig beobachtbar.") 

258 # Dualitätsprinzip: Beobachter ↔ Regler des dualen Systems 

259 result = sig.place_poles(self.A.T, self.C.T, desired_poles) 

260 return result.gain_matrix.T

266 def to_transfer_function(self) -> "TransferFunction": 

267 """Konvertiert in Übertragungsfunktion (nur SISO).""" 

268 from pylab.core.transfer_function import TransferFunction as _TF 

269 num, den = sig.ss2tf(self.A, self.B, self.C, self.D) 

270 return _TF(num[0], den, dt=self.dt, name=self.name)

272 def discretize( 

273 self, dt: float, method: str = "zoh" 

274 ) -> "StateSpace": 

275 """Diskretisiert das kontinuierliche System. 

276 

277 Parameters 

278 ---------- 

279 dt : Abtastzeit [s]. 

280 method : 'zoh' – Zero-Order-Hold (Standard) 

281 'euler' – Euler-Vorwärtsintegration 

282 'bilinear' – Tustin 

283 """ 

284 if self.dt is not None: 

285 raise ValueError("System ist bereits zeitdiskret.") 

286 sys_d = sig.cont2discrete( 

287 (self.A, self.B, self.C, self.D), dt=dt, method=method 

288 ) 

289 Ad, Bd, Cd, Dd = sys_d[:4] 

290 return StateSpace(Ad, Bd, Cd, Dd, dt=dt, name=self.name + "_d")

296 def _sim_time( 

297 self, 

298 t_end: Optional[float], 

299 dt: Optional[float], 

300 ) -> NDArray: 

301 ev = np.abs(np.real(self.eigenvalues())) 

302 ev_pos = ev[ev > 0] 

303 tau = 1.0 / np.min(ev_pos) if len(ev_pos) else 5.0 

304 t_end = t_end or min(max(10 * tau, 1.0), 200.0) 

305 dt = dt or t_end / 2000 

306 return np.linspace(0, t_end, int(t_end / dt) + 1)
