46    name : Bezeichnung.

54    def transfer_function(self) -> TransferFunction:
55        """Gibt die kontinuierliche Übertragungsfunktion C(s) zurück."""
56        Kp, Ti, Td, N = self.Kp, self.Ti, self.Td, self.N
57
58        # P-Regler
59        if Ti == 0 and Td == 0:
60            return TransferFunction([Kp], [1], name=self.name)
61
62        # PI-Regler
63        if Td == 0:
64            # C(s) = Kp*(1 + 1/(Ti*s)) = Kp*(Ti*s + 1)/(Ti*s)
65            return TransferFunction([Kp * Ti, Kp], [Ti, 0], name=self.name)
66
67        # D-Filter: 1/(1 + Td/N * s)
68        d_filter_den = [Td / N, 1.0]
69
70        if Ti == 0:
71            # PD: C(s) = Kp*(1 + Td*s) / (1 + Td/N*s)
72            num = np.polymul([Kp * Td, Kp], [1])
73            den = d_filter_den
74            # Normalisierung
75            num, den = np.polymul(num, [1]), np.polymul(den, [1])
76            return TransferFunction(num, den, name=self.name)
77
78        # Vollständiger PID:  C(s) = Kp*(Ti*Td*s² + Ti*s + 1) / (Ti*s*(1+Td/N*s))
79        # Zähler = Kp*(Ti*Td*s² + Ti*s + 1)
80        pid_num = [Kp * Ti * Td, Kp * Ti, Kp]
81        # Nenner = Ti * [Td/N, 1] * s = Ti * [Td/N, 1, 0]
82        pid_den = np.polymul([Ti, 0], d_filter_den)
83
84        # Zähler ebenfalls mit s multiplizieren (schon im Nenner durch *s)
85        # => C(s) = pid_num / pid_den
86        return TransferFunction(pid_num, pid_den, name=self.name)

141        """Setzt alle internen Zustände zurück."""

143        self._prev_error: float = 0.0
144        self._prev_deriv: float = 0.0

147    def update(self, setpoint: float, measurement: float) -> float:
148        """Berechnet einen Regelschritt und gibt die Stellgröße zurück.
149
150        Parameters
151        ----------
152        setpoint    : Sollwert.
153        measurement : Istwert (gemessener Ausgang).
154
155        Returns
156        -------
157        u : Stellgröße (geclippt auf [u_min, u_max]).
158        """
159        error = setpoint - measurement
160
161        # --- Proportionalanteil ---
162        P = self.Kp * error
163
164        # --- Integralanteil ---
165        if self.Ti != float("inf") and self.Ti > 0:
166            self._integral += error * self.dt
167            I = self.Kp / self.Ti * self._integral
168        else:
169            I = 0.0
170
171        # --- Differentialanteil (mit Tiefpassfilter) ---
172        if self.Td > 0:
173            alpha = self.Td / (self.Td + self.N * self.dt)
174            deriv = alpha * self._prev_deriv + (1 - alpha) * (error - self._prev_error) / self.dt
175            D = self.Kp * self.Td * deriv
176            self._prev_deriv = deriv
177        else:
178            D = 0.0
179
180        u_raw = P + I + D
181        u = float(np.clip(u_raw, self.u_min, self.u_max))
182
183        # --- Anti-Windup ---
184        if self.anti_windup and self.Ti > 0 and self.Ti != float("inf"):
185            windup = (u - u_raw) / (self.Kp / self.Ti) if self.Kp != 0 else 0.0
186            self._integral += windup * self.dt
187
188        self._prev_error = error
189        self._prev_u = u
190        return u

242
243    Das Streckenmodell wird als PTt-Glied (FOPDT) angenähert:
244      G_s(s) ≈ K · e^{-Ls} / (1 + Ts)
245
246    Parameters
247    ----------
248    K             : Statische Streckenverstärkung.

254    -------

256    """

260    elif controller_type == "PI":
