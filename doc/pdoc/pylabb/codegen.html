def gen_digital_filter( G: pylabb.core.TransferFunction, dt: float, method: str = 'tustin', class_name: str = 'DigitalFilter', cfg: Optional[CodegenConfig] = None) -> str: 
258def gen_digital_filter(
259    G: TransferFunction,
260    dt: float,
261    method: str = "tustin",
262    class_name: str = "DigitalFilter",
263    cfg: Optional[CodegenConfig] = None,
264) -> str:
265    """Generiert einen MicroPython-Direktform-II-Filter (diskrete ÜTF).
266
267    Die kontinuierliche ÜTF wird zuerst diskretisiert, dann als
268    Direct-Form-II-Transponiert-Struktur implementiert.
269
270    Parameters
271    ----------
272    G          : Kontinuierliche Übertragungsfunktion.
273    dt         : Abtastzeit [s].
274    method     : Diskretisierungsmethode ('tustin', 'zoh', 'euler').
275    class_name : Name der generierten Klasse.
276    cfg        : CodegenConfig.
277    """
278    cfg = cfg or CodegenConfig()
279    G_d = G.discretize(dt, method=method)
280
281    b_coeffs = G_d.num.real.tolist()
282    a_coeffs = G_d.den.real.tolist()
283
284    # Normierung auf a[0] = 1
285    a0 = a_coeffs[0]
286    b_coeffs = [x / a0 for x in b_coeffs]
287    a_coeffs = [x / a0 for x in a_coeffs]
288
289    order = len(a_coeffs) - 1
290
291    code = _CodeBuilder(cfg.indent)
292    _write_file_header(code, cfg, f"Digitaler Filter (DF-II) – {G.name}")
293
294    code.blank()
295    if cfg.add_comments:
296        code.comment(f"Diskrete ÜTF aus: {G!r}")
297        code.comment(f"Diskretisierung: {method}, dt={dt:.6g}s")
298        code.comment(f"b = {[round(x,8) for x in b_coeffs]}")
299        code.comment(f"a = {[round(x,8) for x in a_coeffs]}")
300        code.blank()
301
302    code.line(f"class {class_name}:")
303    code.indent()
304    code.line("def __init__(self):")
305    code.indent()
306    b_str = "[" + ", ".join(f"{v:.8g}" for v in b_coeffs) + "]"
307    a_str = "[" + ", ".join(f"{v:.8g}" for v in a_coeffs) + "]"
308    code.line(f"self.b = {b_str}  # Zählerkoeffizienten (diskret)")
309    code.line(f"self.a = {a_str}  # Nennerkoeffizienten (diskret)")
310    code.line(f"self._w = [0.0] * {order}  # Verzögerungsspeicher")
311    code.dedent()
312    code.blank()
313
314    code.line("def reset(self):")
315    code.indent()
316    code.line(f"self._w = [0.0] * {order}")
317    code.dedent()
318    code.blank()
319
320    code.line("def update(self, x):")
321    code.indent()
322    if cfg.add_comments:
323        code.comment("Direct-Form-II transponiert")
324        code.comment("x: Eingangswert, Rückgabe: Ausgangswert")
325    code.line("b = self.b")
326    code.line("a = self.a")
327    code.line("w = self._w")
328    code.line("v = x")
329    for i in range(1, order + 1):
330        code.line(f"v -= a[{i}] * w[{i-1}]" if i <= order else "")
331    # Direct Form II transposed implementation
332    code.line("# DF-II transponierte Berechnung")
333    code.line("w_in = x")
334    for i in range(1, min(order + 1, len(a_coeffs))):
335        code.line(f"w_in -= a[{i}] * w[{i-1}]" if i - 1 < order else "")
336    code.line(f"y = b[0] * w_in" + (" + w[0]" if order > 0 else ""))
337    if order > 0:
338        code.line("# Zustands-Update")
339        for i in range(order - 1):
340            code.line(f"w[{i}] = b[{i+1}] * w_in - a[{i+1}] * w_in + w[{i+1}]"
341                      if i + 1 < order else f"w[{i}] = b[{i+1}] * w_in")
342        code.line(f"w[{order-1}] = b[{order}] * w_in" if order < len(b_coeffs)
343                  else f"w[{order-1}] = 0.0")
344    code.line("return y")
345    code.dedent()
346    code.dedent()
347
348    return code.build()
