 60    def rise_time(self, low: float = 0.1, high: float = 0.9) -> float:
 61        """Anstiegszeit von *low* auf *high* des Endwerts (Sprungantwort)."""
 62        yf = self.y[-1]
 63        if abs(yf) < 1e-12:
 64            return float("nan")
 65        times_low = self.t[self.y >= low * yf]
 66        times_high = self.t[self.y >= high * yf]
 67        if not len(times_low) or not len(times_high):  # pragma: no cover
 68            return float("nan")
 69        return float(times_high[0] - times_low[0])
 70
 71    def settling_time(self, band: float = 0.02) -> float:
 72        """Einschwingzeit: letzter Zeitpunkt, an dem y ± band*yf verlässt."""
 73        yf = self.y[-1]
 74        if abs(yf) < 1e-12:
 75            return float("nan")
 76        outside = np.abs(self.y - yf) > band * abs(yf)
 77        if not np.any(outside):
 78            return 0.0
 79        last_outside = np.where(outside)[0][-1]
 80        return float(self.t[last_outside])
 81
 82    def overshoot_pct(self) -> float:
 83        """Überschwingweite in % bezogen auf den Endwert."""
 84        yf = self.y[-1]
 85        if abs(yf) < 1e-12:
 86            return float("nan")
 87        return float(100.0 * (np.max(self.y) - yf) / abs(yf))
 88
 89    def steady_state_error(self) -> float:
 90        """Stationäre Regelabweichung e_∞ = w_∞ − y_∞."""
 91        return float(self.setpoint[-1] - self.y[-1])
 92
 93    def iae(self) -> float:
 94        """Integral Absolute Error ∫|e|dt."""
 95        _trapz = getattr(np, "trapezoid", None) or getattr(np, "trapz")
 96        return float(_trapz(np.abs(self.e), self.t))
 97
 98    def ise(self) -> float:
 99        """Integral Squared Error ∫e²dt."""
100        _trapz = getattr(np, "trapezoid", None) or getattr(np, "trapz")
101        return float(_trapz(self.e**2, self.t))
102
103    def itae(self) -> float:
104        """Integral Time-weighted Absolute Error ∫t|e|dt."""
105        _trapz = getattr(np, "trapezoid", None) or getattr(np, "trapz")
106        return float(_trapz(self.t * np.abs(self.e), self.t))
107
108    def summary(self) -> dict:
109        """Dictionary aller Kennwerte."""
110        return {
111            "Anstiegszeit [s]": self.rise_time(),
112            "Einschwingzeit [s]": self.settling_time(),
113            "Überschwingen [%]": self.overshoot_pct(),
114            "Stationäre Abweichung": self.steady_state_error(),
115            "IAE": self.iae(),
116            "ISE": self.ise(),
117            "ITAE": self.itae(),
118        }

120    def to_signal(self, channel: str = "y") -> Signal:
121        """Gibt einen Kanal als ``Signal``-Objekt zurück."""
122        data = {"y": self.y, "u": self.u, "e": self.e, "setpoint": self.setpoint}
123        return Signal(self.t, data[channel], name=f"{self.name}.{channel}")

130class ClosedLoopSimulator:
131    """Simuliert einen linearen PID-Regelkreis im Zeitbereich.
132
133    Der Regler C wird als diskrete ÜTF oder als
134    ``DiscretePIDController``-Objekt übergeben. Die Strecke G wird
135    entweder als ``TransferFunction``, ``StateSpace`` oder als
136    Callable (nichtlinear!) akzeptiert.
137
138    Beispiel::
139
140        from pylabb.core.transfer_function import TransferFunction
141        from pylabb.control.pid import PIDController, ziegler_nichols_step
142        from pylabb.simulation.time_domain import ClosedLoopSimulator
143        from pylabb.core.signals import step_signal
144
145        G = TransferFunction([1], [1, 2, 1])
146        pid = ziegler_nichols_step(K=1, T=1, L=0.1)
147        C = pid.transfer_function()
148        sim = ClosedLoopSimulator(G, C)
149        result = sim.run(step_signal(t_end=10))
150    """
151
152    def __init__(
153        self,
154        plant: TransferFunction | StateSpace,
155        controller: TransferFunction,
156        disturbance_tf: Optional[TransferFunction] = None,
157        name: str = "Regelkreis",
158    ) -> None:
159        self.plant = plant
160        self.controller = controller
161        self.disturbance_tf = disturbance_tf
162        self.name = name
163
164    def run(
165        self,
166        setpoint: Signal,
167        disturbance: Optional[Signal] = None,
168    ) -> SimulationResult:
169        """Führt die Simulation aus.
170
171        Parameters
172        ----------
173        setpoint    : Sollwertsignal.
174        disturbance : Störsignal auf den Streckeneingang.
175
176        Returns
177        -------
178        SimulationResult
179        """
180        t = setpoint.t
181        w = setpoint.y
182
183        d = disturbance.y if disturbance is not None else np.zeros_like(w)
184
185        # Geschlossener Kreis: T = C*G / (1 + C*G)
186        from pylabb.control.design import closed_loop
187        G = self.plant if isinstance(self.plant, TransferFunction) \
188            else self.plant.to_transfer_function()
189        T = closed_loop(G, self.controller)
190
191        _, y = T.lsim(w, t)
192        e = w - y
193
194        # Stellgröße: u = C * e
195        _, u = self.controller.lsim(e, t)
196
197        # Störantwort (additiv)
198        if self.disturbance_tf is not None:
199            _, d_out = self.disturbance_tf.lsim(d, t)
200            y = y + d_out
201
202        return SimulationResult(
203            t=t, y=y, u=u, e=e, setpoint=w, disturbance=d, name=self.name
204        )
