328    def _on_classify(self) -> None:
329        """Liest Eingaben, ruft classify_loop() auf und zeigt das Ergebnis."""
330        try:
331            g_num = _parse_coeffs(self._g_num.text())
332            g_den = _parse_coeffs(self._g_den.text())
333            c_num = _parse_coeffs(self._c_num.text())
334            c_den = _parse_coeffs(self._c_den.text())
335        except ValueError as exc:
336            QMessageBox.warning(self, "Eingabefehler", f"Ungültige Koeffizienten:\n{exc}")
337            return
338
339        plant = TransferFunction(g_num, g_den, name="G")
340        controller = TransferFunction(c_num, c_den, name="C")
341
342        try:
343            result = classify_loop(
344                plant,
345                controller,
346                has_feedforward=self._cb_feedforward.isChecked(),
347                has_cascade=self._cb_cascade.isChecked(),
348                is_full_state_feedback=self._cb_fsf.isChecked(),
349                is_mimo=self._cb_mimo.isChecked(),
350                has_hysteresis=self._cb_hysteresis.isChecked(),
351                hysteresis_band=self._sb_hysteresis_band.value(),
352                has_disturbance_channel=self._cb_disturbance.isChecked(),
353                threshold=self._sb_threshold.value(),
354            )
355        except Exception as exc:  # noqa: BLE001
356            QMessageBox.critical(self, "Klassifikationsfehler", str(exc))
357            return
358
359        self._last_result = result
360        self._display_result(result)
361        self.classification_done.emit(result)

377        chain_parts = [c.__name__ if hasattr(c, "__name__") else str(c)

389            f"<b>Integrator im Regler:</b> {'ja' if result.controller_has_integrator else 'nein'}",
390            f"<b>Differenzierer im Regler:</b> {'ja' if result.controller_has_derivative else 'nein'}",

392            f"<b>Zusammenfassung:</b> {result.summary}",
393        ]
394        if extensions:
395            lines += ["", "<b>Erweiterungsvorschläge:</b>"]
