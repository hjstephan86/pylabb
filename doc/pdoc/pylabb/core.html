111    A: np.ndarray,
112    B: np.ndarray,
113    dt: float,
114) -> tuple[np.ndarray, np.ndarray]:
115    """Diskretisiert ein zeitkontinuierliches Zustandsraumsystem
116    ẋ = Ax + Bu per Zero-Order-Hold.
117
118    Parameters
119    ----------
120    A  : Systemmatrix (n×n).
121    B  : Eingangsmatrix (n×m).
122    dt : Abtastzeit [s].
123
124    Returns
125    -------
126    Ad, Bd : Diskrete Systemmatrix und Eingangsmatrix.
127
128    Notes
129    -----
130    Verwendet :func:`scipy.linalg.expm` für die Matrixexponentialfunktion.
131    """
132    from scipy.linalg import expm
133
134    n = A.shape[0]
135    m = B.shape[1] if B.ndim == 2 else 1
136
137    # Erweiterungsmatrix für genaue ZOH-Berechnung
138    M = np.zeros((n + m, n + m))
139    M[:n, :n] = A * dt
140    M[:n, n:] = B * dt
141    eM = expm(M)
142
143    Ad = eM[:n, :n]
144    Bd = eM[:n, n:]
145    return Ad, Bd

152def routh_array(coeffs: Sequence[float]) -> np.ndarray:
153    """Berechnet das Routh-Array für das charakteristische Polynom.
154
155    Parameters
156    ----------
157    coeffs : Koeffizienten in absteigender Reihenfolge (a_n, …, a_0).
158
159    Returns
160    -------
161    routh : 2D-Array (n+1 Zeilen).
162
163    Raises
164    ------
165    ValueError : Bei inkonsistenter Eingabe.
166
167    Examples
168    --------
169    >>> routh_array([1, 2, 3, 4])   # s^3 + 2s^2 + 3s + 4
170    """
171    coeffs = np.array(coeffs, dtype=float)
172    n = len(coeffs)
173
174    rows = n
175    cols = int(np.ceil(n / 2))
176    table = np.zeros((rows, cols))
177
178    # Erste zwei Zeilen aus den Koeffizienten
179    table[0, :len(coeffs[0::2])] = coeffs[0::2]
180    table[1, :len(coeffs[1::2])] = coeffs[1::2]
181
182    eps = 1e-12
183    for i in range(2, rows):
184        for j in range(cols - 1):
185            pivot = table[i - 1, 0]
186            if abs(pivot) < eps:
187                pivot = eps  # Sonderfall: Nullzeile
188            table[i, j] = (
189                table[i - 1, 0] * table[i - 2, j + 1]
190                - table[i - 2, 0] * table[i - 1, j + 1]
191            ) / pivot
192
193    return table
